### ⚠️ ВАЖНОЕ ПРИМЕЧАНИЕ

**Поддомены на хостинге (Beget и др.) НЕ ПОДХОДЯТ для парсинга!**

Причины:
1. Все поддомены имеют один IP хостинга - нет ротации
2. IP хостинга быстро заблокируют китайские/корейские сайты
3. Поддомены нужны для входящих запросов, а не для исходящих

**Решение:** Покупайте Residential/Mobile прокси у специализированных провайдеров (Smartproxy, Oxylabs, Bright Data).
Они предоставляют готовые proxy-серверы с ротацией IP и гео-таргетингом.

См. документ: `ПРОКСИ_РЕШЕНИЕ.md`

### Стратегия ротации прокси

```python
# Псевдокод стратегии
PROXY_STRATEGY = {
    "che168.com": {
        "primary": "residential_cn",      # Китайские резидентские
        "fallback": "mobile_asia",        # Мобильные Азия
        "rotation": "per_request",        # Каждый запрос
        "timeout": 30                     # сек
    },
    "dongchedi.com": {
        "primary": "residential_cn",
        "fallback": "mobile_asia",
        "rotation": "per_session",        # Сессия 5-10 запросов
        "timeout": 30
    },
    "encar.com": {
        "primary": "residential_kr",      # Корейские резидентские
        "fallback": "mobile_asia",
        "rotation": "per_session",
        "timeout": 45                     # Encar строже
    }
}
```

### Обход специфичных защит

**che168.com:**
- Проверка cookies (`userarea`)
- Анализ User-Agent
- Решение: использовать Playwright с реальным браузером

**dongchedi.com:**
- AJAX-запросы с токенами
- Session tracking
- Решение: эмулировать полный browser session

**encar.com:**
- `electronic_securities` cookie
- Click tracking
- Решение: mobile proxy + реальная навигация

## 💾 2. База данных и хранение

### Рекомендуемая СУБД: **PostgreSQL 16+**

**Почему PostgreSQL:**
- ✅ Отличная работа с JSON (JSONB тип)
- ✅ Партиционирование таблиц (partitioning)
- ✅ Full-text search встроенный
- ✅ Лучшая работа с 1M+ записей
- ✅ Расширения (pg_trgm для поиска, postgis для геоданных)

### Схема базы данных

```sql
-- Основная таблица автомобилей
CREATE TABLE listings (
    id BIGSERIAL PRIMARY KEY,
    source_site_id SMALLINT NOT NULL,
    external_id VARCHAR(100) NOT NULL,      -- ID на сайте-источнике
    url TEXT NOT NULL,

    -- Информация об авто
    make VARCHAR(50) NOT NULL,              -- Марка (BMW)
    model VARCHAR(100) NOT NULL,            -- Модель (5 Series)
    year SMALLINT NOT NULL,
    mileage INTEGER,                        -- Пробег (км)
    transmission VARCHAR(20),               -- АКПП/МКПП
    fuel_type VARCHAR(20),                  -- Бензин/Дизель/Электро
    engine_volume DECIMAL(3,1),             -- Объем двигателя
    engine_power INTEGER,                   -- Мощность (л.с.)
    drive_type VARCHAR(20),                 -- Привод
    body_type VARCHAR(30),                  -- Кузов
    color VARCHAR(30),
    doors SMALLINT,

    -- Цены
    price_original DECIMAL(12,2) NOT NULL,  -- Оригинальная цена
    currency_original CHAR(3) NOT NULL,     -- CNY/KRW
    price_usd DECIMAL(12,2),                -- Конвертация в USD
    price_rub DECIMAL(12,2),                -- Конвертация в RUB

    -- Перевод
    title_ru VARCHAR(200),                  -- Заголовок RU
    description_ru TEXT,                    -- Описание RU
    features_ru JSONB,                      -- Особенности RU

    -- Статусы
    status VARCHAR(20) DEFAULT 'active',    -- active/sold/archived
    is_translated BOOLEAN DEFAULT FALSE,
    is_exported BOOLEAN DEFAULT FALSE,

    -- Метаданные
    images_json JSONB,                      -- Массив URL изображений
    location VARCHAR(100),
    seller_info JSONB,

    -- Временные метки
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    last_seen_at TIMESTAMPTZ DEFAULT NOW(),
    sold_at TIMESTAMPTZ,

    -- Уникальность
    CONSTRAINT unique_listing UNIQUE (source_site_id, external_id)
);

-- Партиционирование по дате (для производительности)
CREATE TABLE listings_partitioned (
    LIKE listings INCLUDING ALL
) PARTITION BY RANGE (created_at);

-- Индексы
CREATE INDEX idx_listings_make_model ON listings (make, model);
CREATE INDEX idx_listings_year ON listings (year);
CREATE INDEX idx_listings_price ON listings (price_usd);
CREATE INDEX idx_listings_status ON listings (status);
CREATE INDEX idx_listings_created_at ON listings (created_at);
CREATE INDEX idx_listings_source ON listings (source_site_id);
CREATE INDEX idx_listings_location ON listings USING gin (to_tsvector('russian', location));

-- Таблица сайтов-источников
CREATE TABLE source_sites (
    id SMALLSERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    domain VARCHAR(100) NOT NULL,
    country CHAR(2) NOT NULL,
    language CHAR(2) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    parse_schedule VARCHAR(20) NOT NULL,     -- Cron выражение
    last_parse_at TIMESTAMPTZ
);

-- Таблица истории цен (для аналитики)
CREATE TABLE price_history (
    id BIGSERIAL PRIMARY KEY,
    listing_id BIGINT NOT NULL REFERENCES listings(id),
    price_original DECIMAL(12,2) NOT NULL,
    price_usd DECIMAL(12,2),
    price_rub DECIMAL(12,2),
    recorded_at TIMESTAMPTZ DEFAULT NOW()
);

-- Таблица перевода (кэш)
CREATE TABLE translation_cache (
    id BIGSERIAL PRIMARY KEY,
    source_text TEXT NOT NULL,
    source_language CHAR(2) NOT NULL,
    target_language CHAR(2) NOT NULL,
    translated_text TEXT NOT NULL,
    translation_service VARCHAR(30) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),

    CONSTRAINT unique_translation UNIQUE (source_text, source_language, target_language)
);

-- Таблица задач на перевод
CREATE TABLE translation_queue (
    id BIGSERIAL PRIMARY KEY,
    listing_id BIGINT NOT NULL REFERENCES listings(id),
    fields_to_translate JSONB NOT NULL,     -- Какие поля переводить
    status VARCHAR(20) DEFAULT 'pending',   -- pending/processing/completed/failed
    priority SMALLINT DEFAULT 5,            -- 1-10 (1 = высший)
    created_at TIMESTAMPTZ DEFAULT NOW(),
    started_at TIMESTAMPTZ,
    completed_at TIMESTAMPTZ,
    error_message TEXT
);

-- Справочные данные
INSERT INTO source_sites (name, domain, country, language, parse_schedule) VALUES

```

```
│  ├── Качество: ⭐⭐⭐⭐⭐ (лучшее для CN/KR→RU)              │

│  └── Поддержка: CN, KO, JA → RU, EN                        │
│                                                             │
│  Приоритет 2: Google Cloud Translation API                  │
│  ├── Качество: ⭐⭐⭐⭐                                      │
│  ├── Цена: ~$20 за 1M символов                             │
│  ├── Скорость: 500+ запросов/мин                           │
│  └── Автоопределение языка                                 │
│                                                             │
│  Приоритет 3: LibreTranslate (Self-hosted)                  │
│  ├── Качество: ⭐⭐⭐                                        │

```

### Расчет стоимости перевода

**Предположения:**
- 1,000,000 лотов всего
- Средний текст на лот: ~500 символов (заголовок + описание + характеристики)
- Итого: 500,000,000 символов = 500M символов

**Стоимость:**
| Сервис | Цена за 1M | Итого (500M) | Примечание |
|--------|-----------|--------------|------------|
| DeepL | $20 | ~$10,000 | Лучшее качество |
| Google | $20 | ~$10,000 | Быстрее |
| LibreTranslate | $0 | ~$500 | Только сервер |

### Оптимизация расходов на перевод

```python
# Стратегия умного перевода
TRANSLATION_STRATEGY = {
    # Переводить полностью (DeepL)
    "full_translate": [
        "title",           # Заголовок
        "description",     # Основное описание
    ],

    # Переводить шаблонно (словарь терминов)
    "dictionary_translate": [
        "transmission",    # АКПП → Automatic
        "fuel_type",      # 汽油 → Бензин
        "drive_type",     # 四驱 → Полный привод
        "body_type",      # SUV → Внедорожник
        "color",          # 白色 → Белый
    ],

    # Не переводить (числа, бренды)
    "skip_translate": [
        "make",           # BMW остаётся BMW
        "year",
        "mileage",
        "price",
        "engine_volume",
    ],

    # Кэшировать переводы
    "cache_strategy": {
        "enabled": True,
        "ttl_days": 30,
        "fuzzy_match": True,  # 90%相似 → использовать кэш
    }
}
```

```python
# src/proxy/manager.py
import random
from typing import Dict, List, Optional

class ProxyManager:
    def __init__(self, config: Dict[str, Any]):
        self.providers = {
            'residential': ResidentialProxyProvider(config['residential']),
            'mobile': MobileProxyProvider(config['mobile']),
        }
        self.current_proxy = None
        self.failed_proxies = set()

    def get_proxy(self, proxy_type: str = 'residential') -> Dict[str, str]:
        """Получение прокси с ротацией"""
        provider = self.providers[proxy_type]

        # Исключить_failed прокси
        available = [p for p in provider.get_pool() if p['id'] not in self.failed_proxies]

        if not available:
            # Если все прокси failed, очистить список и попробовать снова
            self.failed_proxies.clear()
            available = provider.get_pool()

        proxy = random.choice(available)
        self.current_proxy = proxy
        return proxy['url']

    def mark_failed(self, proxy_id: str):
        """Отметить прокси как failed"""
        self.failed_proxies.add(proxy_id)

    def mark_success(self, proxy_id: str):
        """Отметить прокси как успешный"""
        self.failed_proxies.discard(proxy_id)
```

```python
# src/translation/service.py
class TranslationService:
    def __init__(self, config: Dict[str, Any]):
        self.deepl_client = DeepLClient(config['deepl_api_key'])
        self.dictionary = AutoDictionary()
        self.cache = TranslationCache(config['cache_ttl'])

    async def translate_listing(self, listing: Dict[str, Any]) -> Dict[str, Any]:
        """Перевод всех полей лота"""
        result = listing.copy()

        # Перевод заголовка (DeepL)
        if listing.get('title'):
            result['title_ru'] = await self._translate_with_cache(
                listing['title'], 'zh' if listing['source'] == 'che168' else 'ko'
            )

        # Перевод описания (DeepL)
        if listing.get('description'):
            result['description_ru'] = await self._translate_with_cache(
                listing['description'], 'zh' if listing['source'] == 'che168' else 'ko'
            )

        # Словарный перевод характеристик
        result['transmission_ru'] = self.dictionary.translate(
            listing.get('transmission'),
            listing['source_lang']
        )
        result['fuel_type_ru'] = self.dictionary.translate(
            listing.get('fuel_type'),
            listing['source_lang']
        )
        # ... другие поля

        result['is_translated'] = True
        return result

    async def _translate_with_cache(self, text: str, source_lang: str) -> str:
        """Перевод с кэшированием"""
        # Проверка кэша
        cached = await self.cache.get(text, source_lang, 'ru')
        if cached:
            return cached

        # Перевод через DeepL
        translated = await self.deepl_client.translate(text, source_lang, 'ru')

        # Сохранение в кэш
        await self.cache.set(text, source_lang, 'ru', translated)

        return translated
```
